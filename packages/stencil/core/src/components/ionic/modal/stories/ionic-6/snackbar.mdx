## Usage

```ts
// snackbar.example.ts

async function openModal() {
  const modal = await modalController.create({
    component: 'snackbar-example',
    cssClass: 'tp-snackbar',
  });

  await modal.present();
}

async function closeModal() {
  await modal.dismiss();
}
```
