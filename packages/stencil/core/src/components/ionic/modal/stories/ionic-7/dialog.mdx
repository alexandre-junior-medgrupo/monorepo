## Usage

```ts
// dialog.example.ts

async function openModal() {
  const modal = await modalController.create({
    component: 'dialog-example',
    cssClass: 'tp-dialog',
  });

  await modal.present();
}

async function closeModal() {
  await modal.dismiss();
}
```
