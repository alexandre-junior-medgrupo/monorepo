#### Bash Script

- [Bash utilities](#)

```sh title=".scripts/lint.sh" showLineNumbers
#!/usr/bin/env bash

source .scripts/utils/constants.sh
source .scripts/utils/functions.sh

function lint {
  local TEXT="Running Prettier"
  local TASK="npm run lint"
  local RUNS_ON="root"
  local FILE=".scripts/bash/lint.sh"
  local COMMAND="npx prettier . --write --cache --cache-location .prettiercache"

  heading "${TEXT}"

  body "${TASK}" "${RUNS_ON}" "${FILE}" "${COMMAND}"

  # runs command
  npx prettier . --write --cache --cache-location .prettiercache

  new_line

  footer "${TEXT}"
}

header "Linting Formatting Rules"
lint
```

## Rollup

```bash showLineNumbers
npm i rollup -D
```

### Configuration

```js title="rollup.config.js" showLineNumbers
export default {
  input: 'src/index.js',
  output: {
    file: 'dist/index.js',
  },
};
```

```json {7} title="package.json" showLineNumbers
{
  "name": "templarios-monorepo-cli",
  "version": "1.0.0-alpha.0",
  "description": "Templários monorepo CLI.",
  "scripts": {
    "lint:prettier": "sh .scripts/lint.sh",
    "build": "rollup --config"
  },
  "devDependencies": {
    "rollup": "^4.9.4"
  }
}
```

Para que possamos utilizar rollup.config.`js` precisamos declarar nosso `package.json` como `module`.

```json {5} title="package.json" showLineNumbers
{
  "name": "templarios-monorepo-cli",
  "version": "1.0.0-alpha.0",
  "description": "Templários monorepo CLI.",
  "type": "module",
  "scripts": {
    "lint": "sh .scripts/lint.sh",
    "build": "rollup --config"
  },
  "devDependencies": {
    "rollup": "^4.9.4"
  }
}
```

### TypeScript

- [TypeScript](https://www.typescriptlang.org/)
- [@rollup/plugin-typescript](https://www.npmjs.com/package/@rollup/plugin-typescript)

```bash
npm i typescript @rollup/plugin-typescript -D
```

##### Configuration

```bash
npx tsc --init
```

```json title="tsconfig.json" showLineNumbers
{
  "compilerOptions": {
    "target": "es2016",
    "module": "commonjs",
    "esModuleInterop": true,
    "forceConsistentCasingInFileNames": true,
    "strict": true,
    "skipLibCheck": true
  }
}
```

```js {8} title="rollup.config.js" showLineNumbers
import typescript from '@rollup/plugin-typescript';

export default {
  input: 'src/index.js',
  output: {
    file: 'dist/index.js',
  },
  plugins: [typescript({ tsconfig: './tsconfig.json' })],
};
```
